// TEMPORARY Firestore Security Rules - For Development Only
// Allows all reads and writes for testing
// Replace with stricter rules before going to production

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Allow all reads to everyone (development only)
    match /{document=**} {
      allow read;
    }
    
    // Allow all writes to everyone (development only)
    // In production, this should be restricted by role and authentication
    match /{document=**} {
      allow write;
    }
  }
}
